<div id="result"></div>

<script>
let data = JSON.parse(localStorage.getItem("customers") || "{}");

function save(){
localStorage.setItem("customers", JSON.stringify(data));
}

function search(){
let phone = document.getElementById("phone").value;

if(!data[phone]){
data[phone] = {visits:0};
}

show(phone);
}

function show(phone){
let c = data[phone];

document.getElementById("result").innerHTML = `
<div class='card'>
<p>📞 ${phone}</p>
<p>⭐ الزيارات: ${c.visits}</p>
<button onclick="add('${phone}')">+1 زيارة</button>
</div>`;
}

function add(phone){
data[phone].visits++;
save();
show(phone);
}
</script>
